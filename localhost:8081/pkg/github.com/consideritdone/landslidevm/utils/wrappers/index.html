Packer packs and unpacks a byte array from/to standard values 

type Packer struct {
    Errs

    // The largest allowed size of expanding the byte array
    MaxSize int
    // The current byte array
    Bytes []byte
    // The offset that is being written to in the byte array
    Offset int
}
